   377→        // Nueva función para manejar el switch moderno y sus efectos
   378→        function toggleSectionModern(section) {
   379→            const chk = document.getElementById(`chk-${section}`);
   380→            const container = document.getElementById(`container-${section}`);
   381→            
   382→            // Efectos visuales para las tarjetas de lealtad

   384→                 const card = document.getElementById(`card-${section}`);
   385→                 if(chk.checked) {
   386→                     card.classList.add('active');
   387→                 } else {
   388→                     card.classList.remove('active');
   389→                 }
   390→            }
   391→
   392→            if (chk.checked) {
   393→                container.classList.add('open');
   394→                // Pequeño retraso para dar foco después de la animación
   395→                if (section !== 'experto') {
   396→                    setTimeout(() => {
   397→                        document.getElementById(`inp-${section}`).focus();

   399→                }

   404→                    document.getElementById('inp-promedio').value = '';

   410→                calcular();

   428→        function calcular() {
   429→            const principal = parseFloat(document.getElementById('inp-principal').value) || 0;
   430→            const tasaAnual = parseFloat(document.getElementById('inp-tasa').value) || 0;
   431→            const cargos = parseFloat(document.getElementById('inp-cargos').value) || 0;
   432→            
   433→            // Solo tomar valores si el switch está activo
   434→            const cuotaT0 = document.getElementById('chk-tasa0').checked ? (parseFloat(document.getElementById('inp-tasa0').value) || 0) : 0;
   435→            const cuotaMC = document.getElementById('chk-minicuotas').checked ? (parseFloat(document.getElementById('inp-minicuotas').value) || 0) : 0;
   436→            const totalCuotas = cuotaT0 + cuotaMC;
   437→
   438→            const expertoActivo = document.getElementById('chk-experto').checked;
   439→            const inpPromedio = document.getElementById('inp-promedio').value;
   440→            const promedio = (expertoActivo && inpPromedio !== '') ? parseFloat(inpPromedio) : principal;
   441→            
   442→            const sobregiro = expertoActivo ? (parseFloat(document.getElementById('inp-sobregiro').value) || 0) : 0;
   443→            const mora = expertoActivo ? (parseFloat(document.getElementById('inp-mora').value) || 0) : 0;
   444→            const obligacionesExtra = sobregiro + mora;
   445→
   446→            const tasaMensual = (tasaAnual / 100) / 12;
   447→            const intereses = promedio * tasaMensual;
   448→            let amortizacion = principal / 60; 
   449→            
   450→            const piso = config[moneda].pisoMinimo;
   451→            let subtotalMinimo = amortizacion + intereses + cargos;
   452→            const badgePiso = document.getElementById('badge-piso');
   453→            
   454→            if (principal > 0 && subtotalMinimo < piso && subtotalMinimo > 0) {
   455→                const diferencia = piso - subtotalMinimo;
   456→                amortizacion += diferencia; 
   457→                subtotalMinimo = piso;
   458→                badgePiso.classList.remove('hidden');
   459→            } else {
   460→                badgePiso.classList.add('hidden');
   461→            }
   462→
   463→            const pagoMinimoTotal = subtotalMinimo + totalCuotas + obligacionesExtra;
   464→            const pagoContadoTotal = principal + intereses + cargos + totalCuotas + obligacionesExtra;
   465→
   466→            document.getElementById('res-minimo').innerText = formato(pagoMinimoTotal);
   467→            document.getElementById('res-contado').innerText = formato(pagoContadoTotal);
   468→            
   469→            document.getElementById('res-amortizacion').innerText = formato(amortizacion);
   470→            document.getElementById('res-intereses').innerText = formato(intereses);
   471→            document.getElementById('res-cargos').innerText = formato(cargos);
   472→            document.getElementById('res-cuotas').innerText = formato(totalCuotas);
   473→
   474→            const rowSobregiro = document.getElementById('row-sobregiro');
   475→            if (obligacionesExtra > 0) {
   476→                rowSobregiro.classList.remove('hidden');
   477→                document.getElementById('res-sobregiro').innerText = formato(obligacionesExtra);
   478→            } else {
   479→                rowSobregiro.classList.add('hidden');
   480→            }
   481→
   482→            animateValue('res-minimo');
   483→            animateValue('res-contado');
   484→        }
   485→
   486→        window.onload = calcular;
