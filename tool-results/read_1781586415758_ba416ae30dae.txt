    97→// ============= FORMAT HELPERS =============
    98→function fmtCRC(n: number, cur: string = 'CRC') {
    99→  return new Intl.NumberFormat('es-CR', {
   100→    style: 'currency', currency: cur, minimumFractionDigits: 2, maximumFractionDigits: 2,
   101→  }).format(n)
   102→}
   103→function fmtCompact(n: number): string {
   104→  if (Math.abs(n) >= 1000000) return `${(n / 1000000).toFixed(1)}M`
   105→  if (Math.abs(n) >= 1000) return `${(n / 1000).toFixed(0)}K`
   106→  return n.toFixed(0)
   107→}

   109→// ============= HEALTH SCORE HELPERS =============
   110→function calcHealthScore(params: {
   111→  totalIncome: number; totalExpenses: number; totalDebt: number;
   112→  emergencyFund: number; monthlyEmergencyTarget: number; savingsRate: number
   113→}): number {
   114→  let score = 50
   115→  if (params.savingsRate >= 20) score += 25
   116→  else if (params.savingsRate >= 10) score += 15
   117→  else if (params.savingsRate > 0) score += 5
   118→  else score -= 10
   119→  const debtRatio = params.totalIncome > 0 ? params.totalDebt / params.totalIncome : 0
   120→  if (debtRatio <= 0.2) score += 25
   121→  else if (debtRatio <= 0.35) score += 15
   122→  else if (debtRatio <= 0.5) score += 5
   123→  else score -= 15
   124→  if (params.emergencyFund >= params.monthlyEmergencyTarget) score += 25
   125→  else if (params.emergencyFund >= params.monthlyEmergencyTarget * 0.5) score += 12
   126→  else score -= 5
   127→  const expenseRatio = params.totalIncome > 0 ? params.totalExpenses / params.totalIncome : 1
   128→  if (expenseRatio <= 0.6) score += 25
   129→  else if (expenseRatio <= 0.8) score += 15
   130→  else if (expenseRatio <= 1) score += 0
   131→  else score -= 25
   132→  return Math.max(0, Math.min(100, score))
   133→}

   155→// ============= FINANCIAL CALC HELPERS =============
   156→function calcPMT(principal: number, annualRate: number, months: number): number {
   157→  if (annualRate === 0) return principal / months
   158→  const r = annualRate / 12
   159→  const factor = Math.pow(1 + r, months)
   160→  return principal * (r * factor) / (factor - 1)
   161→}
   162→function amortizationSchedule(principal: number, annualRate: number, months: number) {
   163→  const schedule: { month: number; payment: number; principal: number; interest: number; balance: number }[] = []
   164→  const r = annualRate / 12
   165→  const pmt = calcPMT(principal, annualRate, months)
   166→  let bal = principal
   167→  for (let i = 1; i <= Math.min(months, 60); i++) {
   168→    const int = bal * r
   169→    const prin = pmt - int
   170→    bal = Math.max(0, bal - prin)
   171→    schedule.push({ month: i, payment: pmt, principal: prin, interest: int, balance: bal })
   172→    if (bal === 0) break
   173→  }
   174→  return schedule
   175→}
   176→function payoffTime(principal: number, annualRate: number, payment: number) {
   177→  const r = annualRate / 12
   178→  let bal = principal, totalInt = 0, months = 0
   179→  while (bal > 0 && months < 600) {
   180→    const int = bal * r
   181→    if (payment <= int) return { months: -1, totalInterest: Infinity, totalPaid: Infinity }
   182→    totalInt += int
   183→    bal = bal + int - payment
   184→    months++
   185→    if (bal < 0) bal = 0
   186→  }
   187→  return { months, totalInterest: totalInt, totalPaid: principal + totalInt }
   188→}
   189→function projectCCDebt(principal: number, annualRate: number, payment: number, months: number) {
   190→  const proj: { month: number; balance: number; interestAccum: number; totalPaid: number }[] = []
   191→  const r = annualRate / 12
   192→  let bal = principal, intAccum = 0, paid = 0
   193→  for (let i = 1; i <= months; i++) {
   194→    const int = bal * r
   195→    intAccum += int
   196→    bal = bal + int - payment
   197→    paid += payment
   198→    if (bal < 0) { paid += bal; bal = 0 }
   199→    proj.push({ month: i, balance: bal, interestAccum: intAccum, totalPaid: paid })
   200→    if (bal === 0) break
   201→  }
   202→  return proj
   203→}
   204→function calcNetSalary(gross: number) {
   205→  const ccss = gross * 0.1067
   206→  const taxable = gross - ccss
   207→  let ir = 0
   208→  if (taxable > 4238000) ir = taxable * 0.25 - 272000
   209→  else if (taxable > 2120000) ir = taxable * 0.20 - 60000
   210→  else if (taxable > 1317000) ir = taxable * 0.15 - 25800
   211→  else if (taxable > 922000) ir = taxable * 0.10
   212→  ir = Math.max(0, ir)
   213→  return { gross, ccss, ir, net: gross - ccss - ir, deductions: [{ name: 'CCSS (Seguro Social)', amount: ccss }, { name: 'Impuesto Renta', amount: ir }] }
   214→}
   215→function calcAguinaldoLocal(salaries: number[]) {
   216→  const totalGross = salaries.reduce((s, v) => s + v, 0)
   217→  const monthsWorked = salaries.filter(s => s > 0).length || 1
   218→  const aguinaldoAmount = totalGross / 12
   219→  return { salaries, totalGross, averageSalary: totalGross / monthsWorked, aguinaldoAmount, monthsWorked }
   220→}
