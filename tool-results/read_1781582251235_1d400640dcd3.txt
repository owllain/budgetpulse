   448→   448→        function calcular() {
   449→   449→            const principal = parseFloat(document.getElementById('inp-principal').value) || 0;
   450→   450→            const tasaAnual = parseFloat(document.getElementById('inp-tasa').value) || 0;
   451→   451→            const cargos = parseFloat(document.getElementById('inp-cargos').value) || 0;
   452→   452→            
   453→   453→            // Solo tomar valores si el switch está activo

   457→   457→
   458→   458→            const expertoActivo = document.getElementById('chk-experto').checked;
   459→   459→            const inpPromedio = document.getElementById('inp-promedio').value;
   460→   460→            const promedio = (expertoActivo && inpPromedio !== '') ? parseFloat(inpPromedio) : principal;
   461→   461→            
   462→   462→            const sobregiro = expertoActivo ? (parseFloat(document.getElementById('inp-sobregiro').value) || 0) : 0;
   463→   463→            const mora = expertoActivo ? (parseFloat(document.getElementById('inp-mora').value) || 0) : 0;
   464→   464→            const obligacionesExtra = sobregiro + mora;
   465→   465→
   466→   466→            const tasaMensual = (tasaAnual / 100) / 12;
   467→   467→            const intereses = promedio * tasaMensual;
   468→   468→            let amortizacion = principal / 60; 
   469→   469→            
   470→   470→            const piso = config[moneda].pisoMinimo;
   471→   471→            let subtotalMinimo = amortizacion + intereses + cargos;
   472→   472→            const badgePiso = document.getElementById('badge-piso');
   473→   473→            
   474→   474→            if (principal > 0 && subtotalMinimo < piso && subtotalMinimo > 0) {
   475→   475→                const diferencia = piso - subtotalMinimo;
   476→   476→                amortizacion += diferencia; 
   477→   477→                subtotalMinimo = piso;
   478→   478→                badgePiso.classList.remove('hidden');
   479→   479→            } else {
   480→   480→                badgePiso.classList.add('hidden');
   481→   481→            }
   482→   482→
   483→   483→            const pagoMinimoTotal = subtotalMinimo + totalCuotas + obligacionesExtra;
   484→   484→            const pagoContadoTotal = principal + intereses + cargos + totalCuotas + obligacionesExtra;
   485→   485→
   486→   486→            document.getElementById('res-minimo').innerText = formato(pagoMinimoTotal);
   487→   487→            document.getElementById('res-contado').innerText = formato(pagoContadoTotal);

   491→   491→            document.getElementById('res-cargos').innerText = formato(cargos);

   493→   493→
   494→   494→            const rowSobregiro = document.getElementById('row-sobregiro');
